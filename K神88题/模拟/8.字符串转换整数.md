### 思路
一步步处理即可
难点在数字的拼接

```python
class Solution:

    def myAtoi(self, s: str) -> int:

        n=len(s)

        # 处理空格
        i=0
        while i<n and s[i]==" ":
            i+=1

        # 处理正负号
        sign=1
        if i<n and s[i] in "-+":
            if s[i]=="-":
                sign=-1
            i+=1

        # 处理数字
        top=2**31-1
        num=0
        while i<n and s[i].isdigit():
            num=num*10+int(s[i])
            if num>top:
                if sign==1:
                    return top
                else:
                    return -top-1
            i+=1

        return sign*num
```
