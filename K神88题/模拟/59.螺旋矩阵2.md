```python
class Solution:

    def generateMatrix(self, n: int) -> List[List[int]]:

        matrix=[[0 for _ in range(n)] for _ in range(n)]

        x=1

        top=0
        bottom=n-1
        left=0
        right=n-1

        while True:

            for j in range(left,right+1):
                matrix[top][j]=x
                x+=1
            top+=1
            if top>bottom:
                break

            for i in range(top,bottom+1):
                matrix[i][right]=x
                x+=1
            right-=1
            if right<left:
                break

            for j in range(right,left-1,-1):
                matrix[bottom][j]=x
                x+=1
            bottom-=1
            if bottom<top:
                break

            for i in range(bottom,top-1,-1):
                matrix[i][left]=x
                x+=1
            left+=1
            if left>right:
                break

        return matrix
```
