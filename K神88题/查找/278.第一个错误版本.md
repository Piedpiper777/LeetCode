### 思路
同样的二分查找


```python
# The isBadVersion API is already defined for you.

# def isBadVersion(version: int) -> bool:

class Solution:

    def firstBadVersion(self, n: int) -> int:
        left=1
        right=n
        mid=1
        
        while left<=right:
            mid=(left+right)//2
            if isBadVersion(mid)==True:
                right=mid-1
            else:
                left=mid+1
                
        return left
```
