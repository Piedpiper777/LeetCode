### 思路
维护一个辅助最小栈
或者维护一个辅助最小值

1. 辅助栈
```python
class MinStack:

    def __init__(self):
        self.stack=[]
        self.ministack=[]

    def push(self, value: int) -> None:
        self.stack.append(value)
        if not self.ministack or self.ministack[-1]>=value:
            self.ministack.append(value)

    def pop(self) -> None:
        if self.stack.pop()==self.ministack[-1]:
            self.ministack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.ministack[-1]
```

2. 辅助前缀最小值
```python
class MinStack:

    def __init__(self):
        self.stack=[(0,inf)]

    def push(self, value: int) -> None:
        self.stack.append([value,min(value,self.stack[-1][-1])])

    def pop(self) -> None:
        self.stack.pop()        

    def top(self) -> int:
        return self.stack[-1][0]

    def getMin(self) -> int:
        return self.stack[-1][-1]
```

3. 辅助差值
```python
class MinStack:

  

    def __init__(self):

        self.stack=[]
        self.mini=inf

  

    def push(self, value: int) -> None:
        # 压入的是当前元素和原最小值的差值
        self.stack.append(value-self.mini)

        # 从这里开始讨论，最小值是否更新为了新值
        # 判断依据：新的栈顶元素与0的关系，如果大于零说明没有更新最小值，否则更新了最小值
        self.mini=min(self.mini,value)

  

    def pop(self) -> None:
        # 如果栈顶元素压入时没有更新最小值，直接弹出即可，不需要更新最小值，此时栈顶元素大于0
        # 如果更新了最小值需要还原最小值，此时栈顶元素小于0
        # 更新方式，旧的最小值存储在了栈顶元素中，让value（新最小值）-栈顶元素即可
        self.mini-=min(self.stack.pop(),0)  

  

    def top(self) -> int:
        # 同理，如果栈顶元素小于0说明更新了最小值，此时新元素值直接存在了最小值中，返回self.mini
        # 如果栈顶元素大于0说明没有更新最小值，此时需要一定的还原操作：栈顶元素+最小值，即self.stack.pop()+self.mini
        return self.mini+max(0,self.stack[-1])

  

    def getMin(self) -> int:
        return self.mini
```
