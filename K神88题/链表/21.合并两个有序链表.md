### 思路

直接新建一个辅助节点，然后一一比较两个链表的值，因为两个链表本身就是升序的

```python
# Definition for singly-linked list.

# class ListNode:

#     def __init__(self, val=0, next=None):

#         self.val = val

#         self.next = next

class Solution:

    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:

        dum=ListNode(0)
        cur=dum

        while list1 and list2:

            if list1.val<list2.val:
                cur.next=list1
                list1=list1.next
            else:
                cur.next=list2
                list2=list2.next
                
            cur=cur.next

  

        # 如果尾部有残留，要拼接
        if list1:
            cur.next=list1
        if list2:

            cur.next=list2

        return dum.next
```
