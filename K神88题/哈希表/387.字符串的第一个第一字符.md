### 思路
先统计次数，再找
```python
class Solution:

    def firstUniqChar(self, s: str) -> int:

        cnt=Counter(s)

        res_key=None

        for key,val in cnt.items():

            if val==1:
                res_key=key
                break

        for i,x in enumerate(s):
            if x==res_key:
                return i

        return -1
```

或者
```python
class Solution:

    def firstUniqChar(self, s: str) -> int:
```

```python
        cnt=defaultdict()
        for x in s:
            cnt[x]=cnt.get(x,0)+1
            
        for i,x in enumerate(s):
            if cnt[x]==1:
                return i

        return -1
```
