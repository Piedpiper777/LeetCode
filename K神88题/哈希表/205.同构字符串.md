```python
class Solution:

    def isIsomorphic(self, s: str, t: str) -> bool:

        ## 1.整体序列长度相同
        if len(s)!=len(t):
            return False


        ## 2.出现顺序和位置相同
        dic1,dic2={},{}

        for a,b in zip(s,t):
            if (a in dic1 and dic1[a]!=b) or (b in dic2 and dic2[b]!=a):
                return False
            dic1[a]=b
            dic2[b]=a

        return True
```
