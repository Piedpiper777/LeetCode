### 思路

首先可以直接counter两个字符串，相等即可
```python
class Solution:

    def isAnagram(self, s: str, t: str) -> bool:

        cnt1=Counter(s)
        cnt2=Counter(t)
        
        return cnt1==cnt2
```


一次遍历
```python
class Solution:

    def isAnagram(self, s: str, t: str) -> bool:

        if len(s)!=len(t):
            return False

        n=len(s)
        dic=defaultdict()

        for i in range(n):
            dic[s[i]]=dic.get(s[i],0)+1
            dic[t[i]]=dic.get(t[i],0)-1

        for v in dic.values():
            if v!=0:
                return False

        return True
```

两次遍历
```python
class Solution:

    def isAnagram(self, s: str, t: str) -> bool:

        if len(s)!=len(t):
            return False

        n=len(s)
        dic1=defaultdict()
        dic2=defaultdict()

        for i in range(n):
            dic1[s[i]]=dic1.get(s[i],0)+1
            dic2[t[i]]=dic2.get(t[i],0)+1
            
        return dic1==dic2
```
