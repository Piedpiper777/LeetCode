### 题目描述
给你链表的头结点 `head` ，请将其按 **升序** 排列并返回 **排序后的链表** 。
### 思路
（1）思路一：
新建一个空列表，存储各个点，然后排序，按照排序值新建链表，这里还有两种方法：存值和存节点


### 代码：
- 存值
```python
class Solution:

    def sortList(self, head: ListNode) -> ListNode:
        if not head:
            return None

        # 1.新建列表存储值
        l=list()
        cur=head
        while cur:
            l.append(cur.val)
            cur=cur.next
        l.sort()

        # 2.写入新列表
        dum=ListNode()
        pre=dum
        for x in l:
            pre.next=ListNode(x)
            pre=pre.next
    
        return dum.next
```
- 存节点
```python
class Solution:

    def sortList(self, head: ListNode) -> ListNode:
        if not head:
            return None

        # 1.新建列表存储值
        l=list()
        cur=head
        while cur:
            l.append(cur)
            cur=cur.next
        l.sort(key=lambda x: x.val)

        # 2.新建关系
        pre=ListNode()
        res=pre
        for x in l:
            pre.next=x
            pre=pre.next
        pre.next=None

        return res.next
```
- 归并排序（自上而下）
```python
class Solution:

    def sortList(self, head: ListNode) -> ListNode:
        if not head or not head.next:
            return head

        slow,fast=head,head.next
        while fast and fast.next:
            slow=slow.next
            fast=fast.next.next
        mid=slow.next
        slow.next=None

        left,right=self.sortList(head),self.sortList(mid)

        p=dum=ListNode(0)
        while left and right:
            if left.val<right.val:
                p.next=left
                left=left.next
            else:
                p.next=right
                right=right.next
            p=p.next
        if left:
            p.next=left
        else:
            p.next=right

        return dum.next
```
- 归并排序（自下而上）
```python
class Solution:

    def sortList(self, head: ListNode) -> ListNode:
        # 1.统计链表长度
        p0=head
        length=0
        while p0:
            p0=p0.next
            length+=1

        # 2.初始化区间长度
        ls=1
  
        # 3.建立哨兵节点
        res=ListNode(0)
        res.next=head
  
        # 4.定义外层循环
        while  ls<length:
            pre=res
            h=res.next
            
            # 5.定义内层循环
            while h:
                # 5.1取两个合并区间h1和h2
                h1=h
                i=ls
                while i and h:
                    i-=1
                    h=h.next
                if i:
                    break
                    
                h2=h
                i=ls
                while i and h:
                    i-=1
                    h=h.next
                # 5.2 规定区间长度
                c1=ls
                c2=ls-i

                # 5.3 合并
                while c1 and c2:
                    if h1.val<h2.val:
                        pre.next=h1
                        h1=h1.next
                        c1-=1
                    else:
                        pre.next=h2
                        h2=h2.next
                        c2-=1
                    pre=pre.next
                    
                # h1和h2可能是不等长的，所以可能有一个是有剩余的
                if c1:
                    pre.next=h1
                if c2:
                    pre.next=h2

                # 要把pre移动到最后
                while c1>0 or c2>0:
                    pre=pre.next
                    c1-=1
                    c2-=1

                # 把剩的尾巴接上
                pre.next=h
            ls*=2
        return res.next
```
